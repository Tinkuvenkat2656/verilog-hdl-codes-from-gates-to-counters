`timescale 1ns/1ps
module sr_nor_latch_tb;

  wire q, qbar;

  initial begin
    $dumpfile("wave.vcd");
    $dumpvars(0, sr_nor_latch_tb);
  end

    $monitor("time=%0t, s=%b, r=%b, q=%b, qbar=%b", $time, s, r, q, qbar);
  end
  initial begin
    s=0; r=1; #10;
    s=0; r=0; #10;
    s=1; r=0; #10;
    s=1; r=1; #10;
    $finish;
  end
endmodule
