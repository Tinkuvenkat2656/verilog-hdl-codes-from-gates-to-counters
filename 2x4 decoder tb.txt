  wire [3:0] y;
  decoder_2x4 dut (.a(a), .b(b), .y(y));
  initial begin
    $dumpfile("wave.vcd");
    $dumpvars(0, decoder_2x4_tb);
  end

    $monitor("time=%0t, a=%b, b=%b, y=%b", $time, a, b, y);

   a=0; b=0;  #10;
   a=0; b=1;  #10;

    $finish;
  end
