`timescale 1ns/1ps
module sr_ff_tb;
  reg S, R, clk;
  wire Q;
  sr_ff dut (.S(S), .R(R), .clk(clk), .Q(Q));
  initial clk=0;
  always #5 clk=~clk;
  initial begin
    $dumpfile("wave.vcd");
    $dumpvars(0, sr_ff_tb);
  end
    initial begin 
      $monitor("time=%0t, S=%b, R=%b, clk=%b, Q=%b", $time, S, R, clk, Q);
      S=0; R=0; #10;
      S=0; R=1; #10;
      S=1; R=0; #10;
      S=1; R=1; #10;
      $finish;
    end
endmodule
