module nand_gate_tb;
  reg a,b;
  wire y;
  nand_gate dut (.a(a), .b(b), .y(y));
  initial begin
    $dumpfile("wave.vcd");
    $dumpvars(0, nand_gate_tb);
  end
  initial begin
    $monitor("time=%t, a=%b, b=%b, y=%b", $time, a, b, y);
    a=0; b=0; #10;
    a=0; b=1; #10;
    a=1; b=0; #10;
    a=1; b=1; #10;
    $finish;
  end
endmodule
