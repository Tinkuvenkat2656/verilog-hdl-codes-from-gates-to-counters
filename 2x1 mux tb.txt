`timescale 1ns/1ps
module mux_2x1_tb;
  reg a, b, en, sel;
  wire y;
  mux_2x1 dut (.a(a), .b(b), .en(en), .sel(sel), .y(y));
  initial begin
    $dumpfile("wave.vcd");
    $dumpvars(0, mux_2x1_tb);
  end
  initial begin 
    $monitor("time=%0t, a=%b, b=%b, en=%b, sel=%b, y=%b", $time, a, b, en, sel, y);
    en=0; sel=1'bx; a=0; b=1; #10;
    en=1; sel=0; a=1; b=0; #10;
    en=1; sel=1; a=0; b=1; #10;
    $finish;
  end
endmodule
